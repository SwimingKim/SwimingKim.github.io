### 포인터

```cpp
void swap(int* pA, int* pB)
{
    int temp = *pA;
    *pA = *pB;
    *pB = temp;
}

int* pA, a = 5, b = 10;
pA = &a;
cout << "*pA = " << *pA << " a = " << a << "\n";
cout << "pA = " << pA << " &a = " << &a << "\n";
pswap(pA, &a);
cout << "*pA = " << *pA << " a = " << a << "\n";
cout << "pA = " << pA << " &a = " << &a << "\n";
a = b;
cout << "*pA = " << *pA << " a = " << a << "\n";
cout << "pA = " << pA << " &a = " << &a << "\n";
```

### 레퍼런스

```cpp
void swap(int& rA, int& rB)
{
    int temp = rA;
    rA = rB;
    rB = temp;
}

int a = 5, b = 10;
int& rA = a;
cout << "rA = " << rA << " a = " << a << "\n";
cout << "&rA " << &rA << " &a = " << &a << "\n";
rswap(rA, a);
cout << "rA = " << rA << " a = " << a << "\n";
cout << "&rA = " << &rA << " &a = " << &a << "\n";
a = b;
cout << "&rA = " << &rA << " &a = " << &a << "\n";
```

class Car
{
    private:

Car::Car()
{
    color = '검정';
    gas = 10000;
    cout << "자동차 제조\n";
}

void Car::setColorGas(string c, int g)
{
    color = c;
    gas = g;
};
